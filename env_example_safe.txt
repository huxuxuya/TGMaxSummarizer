# Пример файла конфигурации
# Скопируйте этот файл в .env и заполните реальными значениями

# Telegram Bot Token (получите у @BotFather)
TELEGRAM_BOT_TOKEN=your_telegram_bot_token_here

# VK MAX Token (получите в приложении VK MAX)
VK_MAX_TOKEN=your_vk_max_token_here

# AI Provider API Keys
# GigaChat API Key (получите на https://developers.sber.ru/portal/products/gigachat)
GIGACHAT_API_KEY=your_gigachat_api_key_here

# OpenAI API Key (получите на https://platform.openai.com/api-keys)
OPENAI_API_KEY=your_openai_api_key_here

# OpenRouter API Key (получите на https://openrouter.ai/keys)
OPENROUTER_API_KEY=your_openrouter_api_key_here

# Google Gemini API Key (получите на https://makersuite.google.com/app/apikey)
GEMINI_API_KEY=your_gemini_api_key_here

# AI Provider Settings
DEFAULT_AI_PROVIDER=gigachat
FALLBACK_PROVIDERS=chatgpt,gemini
AI_PROVIDER_TIMEOUT=30
AI_PROVIDER_MAX_RETRIES=3

# Reflection Settings
ENABLE_REFLECTION=true
REFLECTION_TIMEOUT=60
AUTO_IMPROVE_SUMMARY=false
